    <script>
    // üö® CONVERSI√ìN AGRESIVA DE CHECKBOXES - VERSI√ìN MEJORADA üö®
    function convertAllCheckboxes() {
      console.log('üîÑ Buscando checkboxes para convertir...');
      
      // Buscar en TODO el documento
      const allElements = document.querySelectorAll('*');
      let convertedCount = 0;
      
      allElements.forEach(element => {
        // Solo procesar elementos que contengan texto de checkbox y no sean inputs
        if (element.tagName !== 'INPUT' && (element.innerHTML.includes('[ ]') || element.innerHTML.includes('[x]'))) {
          
          // Reemplazar [ ] con checkbox
          if (element.innerHTML.includes('[ ]')) {
            element.innerHTML = element.innerHTML.replace(/\[ \]/g, '<input type="checkbox" class="custom-checkbox-js">');
            convertedCount++;
          }
          
          // Reemplazar [x] con checkbox marcado
          if (element.innerHTML.includes('[x]')) {
            element.innerHTML = element.innerHTML.replace(/\[x\]/g, '<input type="checkbox" class="custom-checkbox-js" checked>');
            convertedCount++;
          }
          
          // Agregar clase para styling
          if (element.tagName === 'LI') {
            element.classList.add('task-item-custom-js');
          }
        }
      });
      
      console.log(`‚úÖ Convertidos ${convertedCount} checkboxes`);
    }

    // Ejecutar cuando el DOM est√© listo
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', convertAllCheckboxes);
    } else {
      convertAllCheckboxes();
    }

    // Tambi√©n ejecutar despu√©s de un delay por si el contenido se carga din√°micamente
    setTimeout(convertAllCheckboxes, 1000);
    setTimeout(convertAllCheckboxes, 3000);

    // Observar cambios en el DOM para contenido que se cargue despu√©s
    const observer = new MutationObserver(function(mutations) {
      convertAllCheckboxes();
    });

    observer.observe(document.body, {
      childList: true,
      subtree: true,
      characterData: true
    });
    </script>
