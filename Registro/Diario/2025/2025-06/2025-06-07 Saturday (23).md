// Carpeta donde buscar
const folderPath = "Registro/Diario";

// Encabezado a buscar
const targetHeading = "## #Obra/Zapata";

// Obtenemos todos los archivos de esa carpeta
const files = app.vault.getMarkdownFiles().filter(f => f.path.startsWith(folderPath));

for (let file of files) {
    const content = await app.vault.read(file);
    const lines = content.split("\n");
    let output = [];
    let capture = false;

    for (let line of lines) {
        // Si encontramos el encabezado correcto, comenzamos a capturar
        if (line.trim() === targetHeading.trim()) {
            capture = true;
            continue;
        }

        // Si llegamos a otro encabezado de nivel 2, detenemos la captura
        if (line.startsWith("## ") && !line.startsWith("## #Obras/Zapata")) {
            if (capture) break;
        }

        // Capturamos líneas mientras estamos en la sección
        if (capture && line.trim() !== "") {
            output.push(line);
        }
    }

    // Mostramos el resultado si hay contenido capturado
    if (output.length > 0) {
        dv.header(4, `Contenido en ${file.name}`);
        dv.paragraph(output.join("  \n"));
    }
}
